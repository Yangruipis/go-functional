** Install

#+BEGIN_SRC
go get github.com/Yangruipis/go-functional
#+END_SRC

*** requirements:

- ~go>=1.18~

** Usage

#+BEGIN_SRC go
  package main

  import (
    "fmt"

    fun "github.com/Yangruipis/go-functional/pkg"
  )

  func main() {
    fun.RangeSeq(0, 10, 1).Map(func(k, v int) (int, int) {
      return k, v + 1
    }).Filter(func(k, v int) bool { return v >= 3 }).ForEach(func(i, v int) {
      fmt.Printf("%d\n", v)
    })
  }
#+END_SRC
